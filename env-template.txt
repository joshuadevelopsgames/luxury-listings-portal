# OpenAI API Configuration
# Get your API key from: https://platform.openai.com/api-keys
REACT_APP_OPENAI_API_KEY=your_openai_api_key_here

# Note: Replace 'your_openai_api_key_here' with your actual OpenAI API key
# The REACT_APP_ prefix is required for React to access the environment variable

# To use this:
# 1. Copy this content
# 2. Create a new file called .env.local in your project root
# 3. Paste the content and replace 'your_openai_api_key_here' with your real API key
# 4. Restart your development server
